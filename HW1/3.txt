<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true?    При сравнении числа со строкой, строка преобразуется в численный тип, поэтому происходит сравнение 5 == 5 /true ( 05 преобразовалось в 5, так как целое число не может начинаться с 0);
    var_dump((int)'012345');     // Почему 12345?   Тип integer (целое число) не может начинаться с 0;
    var_dump((float)123.0 === (int)123.0); // Почему false?   === сравнивает типы, в нашем случае float и int не совпадают; 
    var_dump((int)0 === (int)'hello, world'); // Почему true?   Строка 'hello, world' преобразуется в число (так как тип int), оно не содержит чисел, поэтому после преобразования получим 0, 0===0 /true (int совпадают);
?>
